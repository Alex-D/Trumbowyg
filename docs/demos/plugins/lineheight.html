Custom font sizes

Change the font!

The code


$('#editor-custom-font-sizes')
.trumbowyg({
    btns: [
        ['lineheight']
    ],
    plugins: {
        lineheight: {
            sizeList: [
                '14px',
                '18px',
                '22px'
            ]
        }
    }
});
